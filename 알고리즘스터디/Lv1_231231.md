function solution(n, lost, reserve) {
var answer = 0;
reserve = reserve.reverse()
lost = lost.reverse()
reserve.map((num,index) => {
if(reserve.includes(lost[lost.length-1] - 1) || lost[lost.length-1] + 1) {
reserve.pop()
lost.pop()
}
})
answer = n - lost.length

}
