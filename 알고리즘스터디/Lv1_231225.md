   음식 종류 양 같아야 함
   칼로리가 낮은 음식을 먼저 먹을 수 있게 배치
   배치를 하는 것이 문제.

c = food[i] / 2

function solution(food) {
let arr1 = [];
let answer = "";
for(let i = 1; i < food.length; i++) {
let b = food[i]/2
console.log(Math.floor(b))
for(let j = 1; j<=b; j++) {
arr1.push(i)
}
}
let arr2 = [...arr1].reverse()
arr1.map((number) => answer+=number);
answer+=0
arr2.map((number) => answer+=number)
return answer;
}

2. 비밀지도

십진수 -> 2진수 모두 변경
그 후 arr1 과 arr2 둘 중 하나만이라도 1이 있으면, 1을 추가
function solution(n, arr1, arr2) {
var answer = [];
let arr1to = [];
let arr2to = [];
arr1.forEach((number) => arr1to.push(String(number.toString(2).padStart(n,"0"))))
arr2.forEach((number) => arr2to.push(String(number.toString(2).padStart(n,"0"))))
for(let i = 0; i < n; i++) {
let test1 = arr1to[i];
let test2 = arr2to[i];
let t = "";
for(let j = 0; j < n; j++) {
if(test1[j] == 1 || test2[j] == 1) {
t+="#"
}
if(test1[j] == 0 && test2[j] == 0) {
t+=" "
}
}
answer.push(t)
}

    return answer;

}
