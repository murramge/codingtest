function solution(a, b) {
let sum = 0;
for(let i = 0; i<a.length; i++) {
sum += parseInt(a[i]) \* parseInt(b[i])
}
return sum
}

function solution(n) {
var answer = '';
for(let i = 0; i < n; i++) {
i % 2 == 0 ? answer+="수" : answer+="박"
}
return answer;
}
